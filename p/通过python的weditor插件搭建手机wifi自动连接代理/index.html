        if d(text=t).exists:
            d(text=t).click()
            break
    else:
        for rid in BACK_IDS:
            if d(resourceId=rid).exists:
                d(resourceId=rid).click()
                break
        else:
            d.press("back")
    if isGoHome:
        go_home(d, timeout=go_home_timeout)
